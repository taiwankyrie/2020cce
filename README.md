##第一題:進階題：分式化簡 
```C
#include <stdio.h>
int main()
{
	int a,b,c,A,B;
	scanf("%d%d", &a, &b);
	A=a;
	B=b;
	c=a%b;
	while(c!=0)
	{
	 a=b;
	 b=c;
	 c=a%b;
	}
	printf("%d %d\n",A/b,B/b);
}
```
##第二題:進階題：讀入整數反序列印
```C
#include <stdio.h>
int main()
{
	int a[10],b=0;
	for(int i=0;i<10;i++)
	{
		scanf("%d", &a[i]);
		if(a[i]==0) break;
		else b++;
	}
	for(int i=b-1;i>=0;i--)
	{
		printf("%d ",a[i]);
	}
	printf("\n");
}
```

##第三題:進階題：A的B次方函數
```C
#include <stdio.h>
int MYPOWER(int a, int b)
{
	int ans=1;
	for(int i=1;i<=b;i++)
	{
		ans*=a;
		
	}
	return ans;
}

int main(void)
{
	int a,b;
	scanf("%d%d",&a,&b);
	printf("[%d]",MYPOWER(a,b));
	return 0;
}
```
##第四題:進階題：漸增數列相加
```C
#include <stdio.h>
int main()
{
	int a,ans=0;
	scanf("%d", &a);
	for(int i=1;i<=a-1;i++)
	{
		ans+=i*(i+1);
	}
	printf("%d\n",ans);
}
```
